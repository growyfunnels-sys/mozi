e.parameter : {}; const action = params.action || 'get'; const callback = params.callback || null; const dataParam = params.data || null; let result; if (action === 'set' && dataParam) { result = setData(dataParam); } else { result = readData(); } if (callback) { return ContentService .createTextOutput(callback + '(' + JSON.stringify(result) + ');') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ── POST: escritura desde iframe/form ───────────────── function doPost(e) { try { let dataStr = null; // Puede venir como form field o como JSON body if (e.parameter && e.parameter.data) { dataStr = e.parameter.data; } else if (e.postData && e.postData.contents) { const body = JSON.parse(e.postData.contents); dataStr = typeof body.data === 'string' ? body.data : JSON.stringify(body.data); } if (!dataStr) return buildResponse({ error: 'Sin datos' }); const result = setData(dataStr); return buildResponse(result); } catch(err) { return buildResponse({ error: err.toString() }); } } function readData() { try { const sheet = getSheet(); const raw = sheet.getRange(2, 1).getValue(); if (!raw || raw === '{}' || raw === '') { return { prospectos: [], facturacionReal: 0, nextId: 1, rubrosExtra: [], ciudadesExtra: [] }; } const data = JSON.parse(raw); // Asegurar que tenga todas las claves necesarias if (!data.prospectos) data.prospectos = []; if (!data.rubrosExtra) data.rubrosExtra = []; if (!data.ciudadesExtra) data.ciudadesExtra = []; return data; } catch(err) { return { error: err.toString() }; } } function setData(dataStr) { try { const data = JSON.parse(dataStr); const sheet = getSheet(); sheet.getRange(2, 1).setValue(JSON.stringify(data)); SpreadsheetApp.flush(); return { ok: true, savedAt: new Date().toISOString() }; } catch(err) { return { error: err.toString() }; } } function buildResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
